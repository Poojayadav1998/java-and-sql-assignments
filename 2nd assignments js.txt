    <script>
        function print_prime(n){
            let i,j,flag;
            
            for(i=1;i<=n;i++)
            {
                if(i==1 || i==0)
                continue;
                flag = 1;
                for(j=2;j<=i/2;j++){
                    if(i%j == 0){
                        flag=0;
                        break;
                    }
                }
                if(flag==1)
                document.write(i+ "   ")
            }
        }
        let n=50;
        print_prime(n);
    </script>
</body>

  <script>
    let n = 8; 
    let string = "";
    
    for (let i = 1; i <= n; i++) {
      
      for (let j = n; j > i; j--) {
        string += " ";
      }
      
      for (let k = 0; k < i * 2 - 1; k++) {
        string += "*";
      }
      string += "\n";
    }

    for (let i = 1; i <= n - 1; i++) {
      
      for (let j = 0; j < i; j++) {
        string += " ";
      }
    
      for (let k = (n - i) * 2 - 1; k > 0; k--) {
        string += "*";
      }
      string += "\n";
    }
    document.write(`<pre>${string}</pre>`);
  </script>
